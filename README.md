echo -n "$GENERATION_TIME" >> README.md

- **压缩比**: EOF
if [ "$ORIGINAL_LENGTH" -gt 0 ]; then
  RATIO=$(echo "scale=2; $ENCRYPTED_LENGTH / $ORIGINAL_LENGTH" | bc -l 2>/dev/null || echo "N/A")
  echo -n "${RATIO}x" >> README.md
else
  echo -n "N/A" >> README.md
fi
cat >> README.md << 'EOF'
